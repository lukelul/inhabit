# One contract. Whole platform.
from inhabit import RobotAdapter, pvt

class UR10eAdapter(RobotAdapter):
    protocol  = "rtde"
    transport = "webrtc | lan"
    dof       = 6

    def command(self, twist, grip):
        self.rtde.servo(twist, gain=0.85)

    def streams(self):
        return pvt.auto(self)  # P+V+T discovery
